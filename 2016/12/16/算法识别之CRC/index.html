CRC的识别比较简单了。

unsigned long table[256];
//位逆转  
static unsigned long bitrev(unsigned long input, int bw)
{
    int i;
    unsigned long var;
    var = 0;
    for (i = 0; i < bw; i++)
    {
        if (input & 0x01)
        {
            var |= 1 << (bw - 1 - i);
        }
        input >>= 1;
    }
    return var;
}
//码表生成  
//如:X32+X26+...X1+1,poly=(1<<26)|...|(1<<1)|(1<<0)  
void crc32_init(unsigned long poly)
{
    int i;
    int j;
    unsigned long c;
    poly = bitrev(poly, 32);
    for (i = 0; i < 256; i++)
    {
        c = i;
        for (j = 0; j < 8; j++)
        {
            if (c & 1)
            {
                c = poly ^ (c >> 1);
            }
            else
            {
                c = c >> 1;
            }
        }
        table[i] = c;
        printf("CRC32表:%x\n", c);
    }
}
unsigned long crc32(unsigned long crc, void* input, int len)
{
    int i;
    unsigned char index;
    unsigned char* pch;
    pch = (unsigned char*)input;
    for (i = 0; i < len; i++)
    {
        index = (unsigned char)(crc^*pch);
        crc = (crc >> 8) ^ table[index];
        pch++;
    }
    return crc;
}
int _tmain(int argc, _TCHAR* argv[])
{
    unsigned long crc;
    crc32_init(0x4C11DB7);  
    crc = 0xFFFFFFFF;
    crc = crc32(crc, "1234567890", 10);
    crc ^= 0xFFFFFFFF;
    printf("CRC32=%08X\n", crc);
    system("pause");
}

初始化的这张表就是关键了。
